  if (holdings.length === 0) {
    alert("Add some holdings first");
    return;
  }
  
  setReviewLoading(true);
  setReviewResult(null);
  
  try {
    const response = await fetch("/api/portfolio/review", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ holdings }),
    });
    
    const data = await response.json();
    
    if (data.message) {
      setReviewResult(data.message);
    } else if (data.display_type === "chat") {
      setReviewResult(data.message || "No analysis returned");
    } else {
      setReviewResult(JSON.stringify(data, null, 2));
    }
  } catch (err) {
    setReviewResult("Failed to get portfolio review. Please try again.");
  }
  
  setReviewLoading(false);
};
