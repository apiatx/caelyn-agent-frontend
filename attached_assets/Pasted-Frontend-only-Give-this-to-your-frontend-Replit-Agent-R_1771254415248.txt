tsxconst sendMessage = async (query: string) => {
  // Detect if this looks like a watchlist review (multiple tickers)
  const tickerPattern = /^(review|analyze|check|watchlist)?\s*[A-Z]{1,5}(\s*[,\s]\s*[A-Z]{1,5}){2,}/i;
  const cleanedQuery = query.trim();
  
  // If it matches "NVDA, AMD, PLTR" or "Review NVDA AMD PLTR" pattern (3+ tickers)
  if (tickerPattern.test(cleanedQuery)) {
    // Extract tickers
    const tickers = cleanedQuery
      .replace(/^(review|analyze|check|watchlist|my)\s*/i, "")
      .split(/[,\s]+/)
      .map(t => t.trim().toUpperCase())
      .filter(t => /^[A-Z]{1,5}$/.test(t))
      .slice(0, 25);
    
    if (tickers.length >= 3) {
      // Route to watchlist endpoint
      await sendWatchlistReview(tickers);
      return;
    }
  }
  
  // Otherwise, send to normal /api/query endpoint
  await sendNormalQuery(query);
};
