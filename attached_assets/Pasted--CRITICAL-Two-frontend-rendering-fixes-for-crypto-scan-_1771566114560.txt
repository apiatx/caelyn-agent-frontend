    if (rate == null || rate === "" || rate === "N/A") return null;
    if (typeof rate === 'string' && rate.includes('%')) return rate; // Already formatted
    if (typeof rate === 'number') {
        if (Math.abs(rate) < 1) {
            // Raw rate like 0.0029 â€” multiply by 100
            return `${(rate * 100).toFixed(4)}%`;

    return String(rate);
